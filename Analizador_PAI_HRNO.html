// ── Nombre de la hoja destino
const HOJA = 'NIÑOS Y NIÑAS';

// ── Columnas del encabezado consolidado (deben coincidir con el
//    orden en que el app transforma las filas de la plantilla)
const ENCABEZADO = [
  'MUNICIPIO', 'FECHA_VACUNACION', 'CONSECUTIVO', 'DIA',
  'TIPO ID MADRE', 'NUM ID MADRE',
  'PRIMER NOMBRE MADRE', 'SEGUNDO NOMBRE MADRE',
  'PRIMER APELLIDO MADRE', 'SEGUNDO APELLIDO MADRE',
  'TIPO ID NIÑO', 'NUM ID NIÑO', 'FECHA NACIMIENTO',
  'TIPO EDAD', 'EDAD A LA FECHA DE VACUNACION', 'EDAD ACTUAL',
  'SEXO',
  'PRIMER APELLIDO NIÑO', 'SEGUNDO APELLIDO NIÑO',
  'PRIMER NOMBRE NIÑO', 'SEGUNDO NOMBRE NIÑO',
  'REGIMEN', 'ASEGURADORA', 'DEPTO RESIDENCIA',
  'MUNICIPIO RESIDENCIA', 'AREA RESIDENCIA',
  'BARRIO / VEREDA', 'DIRECCION', 'TELEFONO',
  'GRUPO ETNICO', 'CONDICION DESPLAZAMIENTO',
  'CONDICION DISCAPACIDAD', 'CORREO'
];

// ════════════════════════════════════════════════════════════
//  doPost — recibe filas desde el aplicativo HTML
// ════════════════════════════════════════════════════════════
function doPost(e) {
  // Lock de escritura: evita corrupción si llegan dos lotes a la vez
  const lock = LockService.getScriptLock();
  lock.waitLock(30000);

  try {
    const body = JSON.parse(e.postData.contents);
    const ss   = SpreadsheetApp.getActiveSpreadsheet();

    const nombreHoja = (body.hoja || HOJA).toString().trim();
    let sheet = ss.getSheetByName(nombreHoja);

    // Crear la hoja si no existe
    if (!sheet) {
      sheet = ss.insertSheet(nombreHoja);
    }

    const filas = body.rows || [];
    if (filas.length === 0) {
      return ok({ agregados: 0, reemplazados: 0 });
    }

    // ── Crear encabezado si la hoja está vacía
    if (sheet.getLastRow() === 0) {
      const hdrs = (body.headers && body.headers.length > 0)
        ? body.headers
        : ENCABEZADO;
      sheet.appendRow(hdrs);
      sheet.getRange(1, 1, 1, hdrs.length)
        .setFontWeight('bold')
        .setBackground('#1c4966')
        .setFontColor('#ffffff');
      sheet.setFrozenRows(1);
    }

    // ── Deduplicación: si ya existen filas para este municipio+mes,
    //    borrarlas antes de insertar (evita duplicados al reenviar)
    const municipio = body.municipio || '';
    const yearMon   = body.yearMon   || '';   // formato "2026-04"

    if (municipio && yearMon) {
      eliminarFilasExistentes(sheet, municipio, yearMon);
    }

    // ── Inserción en lote con setValues (1 llamada a la API, no N)
    const ultimaFila = sheet.getLastRow();
    const ncols      = filas[0].length;
    sheet
      .getRange(ultimaFila + 1, 1, filas.length, ncols)
      .setValues(filas);

    SpreadsheetApp.flush();
    return ok({ agregados: filas.length });

  } catch (err) {
    return error(err.toString());
  } finally {
    lock.releaseLock();
  }
}

// ════════════════════════════════════════════════════════════
//  doGet — devuelve el estado de carga o info de la hoja
// ════════════════════════════════════════════════════════════
function doGet(e) {
  try {
    const accion = (e.parameter && e.parameter.action) || 'status';

    if (accion === 'status') return getStatus();
    if (accion === 'datos')  return getDatos();
    if (accion === 'info')   return getInfo();

    return error('Accion no reconocida: ' + accion);
  } catch (err) {
    return error(err.toString());
  }
}

// ── Retorna {municipio||YYYY-MM: cantidad} para el grid de estado
function getStatus() {
  const ss    = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(HOJA);

  if (!sheet || sheet.getLastRow() < 2) return ok({});

  const datos   = sheet.getDataRange().getValues();
  const hdrs    = datos[0].map(h => norm(h));
  const colMun  = hdrs.findIndex(h => h.includes('municipio') && !h.includes('resid'));
  const colFech = hdrs.findIndex(h => h.includes('fecha_vac') || h.startsWith('fecha vac'));

  const status = {};
  datos.slice(1).forEach(row => {
    const mun = colMun >= 0 ? row[colMun] : row[0];
    const f   = colFech >= 0 ? row[colFech] : row[1];
    if (!mun || !f) return;

    const yearMon = (f instanceof Date)
      ? Utilities.formatDate(f, 'America/Bogota', 'yyyy-MM')
      : f.toString().substring(0, 7);

    if (yearMon && yearMon.length === 7) {
      const key = mun.toString().trim() + '||' + yearMon;
      status[key] = (status[key] || 0) + 1;
    }
  });

  return ok(status);
}

// ── Retorna todos los datos para el módulo Análisis PAI
function getDatos() {
  const ss    = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(HOJA);

  if (!sheet || sheet.getLastRow() < 2) {
    return ok({ headers: [], rows: [] });
  }

  const all = sheet.getDataRange().getValues();

  // Convertir fechas Date a string "yyyy-mm-dd" para que el JSON sea serializable
  const tz = 'America/Bogota';
  const serialized = all.map(row =>
    row.map(cell => {
      if (cell instanceof Date) {
        return Utilities.formatDate(cell, tz, 'yyyy-MM-dd');
      }
      return cell;
    })
  );

  return ok({
    headers: serialized[0],
    rows:    serialized.slice(1)
  });
}

// ── Retorna metadatos básicos de la hoja (columnas, filas)
function getInfo() {
  const ss    = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(HOJA);
  if (!sheet) return ok({ existe: false });
  return ok({
    existe:   true,
    filas:    sheet.getLastRow() - 1,
    columnas: sheet.getLastColumn(),
    nombre:   sheet.getName()
  });
}

// ════════════════════════════════════════════════════════════
//  UTILIDADES INTERNAS
// ════════════════════════════════════════════════════════════

// Borra todas las filas de un municipio+mes (para reenvío limpio)
function eliminarFilasExistentes(sheet, municipio, yearMon) {
  if (sheet.getLastRow() < 2) return;

  const datos   = sheet.getDataRange().getValues();
  const hdrs    = datos[0].map(h => norm(h));
  const colMun  = hdrs.findIndex(h => h.includes('municipio') && !h.includes('resid'));
  const colFech = hdrs.findIndex(h => h.includes('fecha_vac') || h.startsWith('fecha vac'));

  // Recorrer de atrás hacia adelante para no desplazar índices al borrar
  for (let i = datos.length - 1; i >= 1; i--) {
    const row  = datos[i];
    const mun  = colMun >= 0 ? row[colMun] : row[0];
    const f    = colFech >= 0 ? row[colFech] : row[1];
    const ym   = (f instanceof Date)
      ? Utilities.formatDate(f, 'America/Bogota', 'yyyy-MM')
      : (f || '').toString().substring(0, 7);

    if (mun && mun.toString().trim() === municipio && ym === yearMon) {
      sheet.deleteRow(i + 1); // +1 porque getValues es 0-based, Sheet es 1-based
    }
  }
}

// Normaliza texto para comparaciones (minúscula, sin tildes, sin símbolos)
function norm(s) {
  return (s || '').toString().toLowerCase()
    .normalize('NFD').replace(/[̀-ͯ]/g, '')
    .replace(/[^a-z0-9\s_]/g, ' ').replace(/\s+/g, ' ').trim();
}

// Helpers de respuesta JSON
function ok(data) {
  return ContentService
    .createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}

function error(msg) {
  return ContentService
    .createTextOutput(JSON.stringify({ error: msg }))
    .setMimeType(ContentService.MimeType.JSON);
}
